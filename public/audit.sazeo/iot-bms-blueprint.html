-- ორგანიზაცია = tenant boundary. მონაცემები home-region-ს არასდროს ტოვებენ.
CREATE TABLE organizations (
  id              uuid        PRIMARY KEY,
  name            text        NOT NULL,           -- ორგანიზაციის სახელი
  tier            text        NOT NULL,           -- free | pro | enterprise
  home_region     text        NOT NULL,           -- eu-central-1 | us-east-1 | …
  created_at      timestamptz NOT NULL DEFAULT now()
);

-- user ≠ org-is მოქალაქე default-ად. visibility მხოლოდ explicit grant-ით.
CREATE TABLE users (
  id              uuid        PRIMARY KEY,
  email           citext      NOT NULL UNIQUE,    -- case-insensitive ელფოსტა
  email_verified  boolean     NOT NULL DEFAULT false,
  cognito_sub     text        UNIQUE,              -- Cognito sub-ის ხიდი
  created_at      timestamptz NOT NULL DEFAULT now()
);

-- N:M grant (ADR-010). როლი = org-scope. JWT-ში არ ვდებთ — session-resolved.
CREATE TABLE memberships (
  user_id         uuid        NOT NULL REFERENCES users(id),
  organization_id uuid        NOT NULL REFERENCES organizations(id),
  role            text        NOT NULL,           -- owner | admin | operator | viewer
  granted_at      timestamptz NOT NULL DEFAULT now(),
  PRIMARY KEY (user_id, organization_id)
);

-- RLS ყოველ tenant-scoped ცხრილზე. middleware SET LOCAL app.tenant_id-ს უკეთებს.
ALTER TABLE memberships ENABLE ROW LEVEL SECURITY;
CREATE POLICY memberships_tenant ON memberships
  USING (organization_id = current_setting('app.tenant_id')::uuid);

Telemetry — TimescaleDB hypertable (ADR-007, 008)
-- სენსორის წერტილები. partitioned by time; 7-day chunks; continuous aggregates.
CREATE TABLE telemetry (
  time            timestamptz NOT NULL,           -- სენსორის დროშტამპი (device clock)
  tenant_id       uuid        NOT NULL,
  device_id       uuid        NOT NULL,
  metric          text        NOT NULL,           -- temperature | voltage | flow | pressure
  value           double precision NOT NULL,
  unit            text        NOT NULL,           -- °C | V | m³/h | bar
  quality         smallint    NOT NULL DEFAULT 100, -- 0–100 (sensor confidence)
  gap_ratio       real        NOT NULL DEFAULT 0   -- ADR-015: >0.1 → billing-ში გამოვიცვალოთ
);

SELECT create_hypertable('telemetry', 'time', chunk_time_interval => interval '7 days');

-- საათობრივი CA → feature store (ADR-008). AI/ML Timescale-დან კითხულობს.
CREATE MATERIALIZED VIEW telemetry_1h
WITH (timescaledb.continuous) AS
  SELECT time_bucket('1 hour', time) AS hour,
         tenant_id, device_id, metric,
         avg(value) AS v_avg,
         min(value) AS v_min,
         max(value) AS v_max,
         count(*)   AS samples
  FROM telemetry
  GROUP BY 1, 2, 3, 4;

− Timestream serverless ერთი ქვრივი გამოგვაცვლიდა, მაგრამ SQL-ის ქვესიმცირე არის limiting, migration-ი უფრო რთულია

ჩვენი არჩევანიალტერნატივებირატომ ჩვენი

Revise trigger: >5k concurrent subscribers/node → AppSync migration

როლი და საჭიროება — თითოეული კომპონენტი დეტალურად

ქართული განმარტება რატომ არის საჭირო · რა მოხდება ამის გარეშე · რას ცვლის სისტემის ქცევაში
§24.1 — მონაცემთა ფენა (Data layer)
კომპონენტიროლირატომ საჭიროა (დეტალურად)
Aurora PostgreSQL 16 სისტემის მთავარი ტრანზაქციული ბაზა — აქ ინახება მომხმარებლები, ორგანიზაციები, მოწყობილობების რეესტრი, ბრძანებები, შეტყობინებები ეს არის "source of truth" ყველა ოპერაციისთვის, რომელიც ACID-ს საჭიროებს (ფინანსური ტრანზაქციები, როლის შეცვლა, membership grants). Aurora-ს 6-ჯერ რეპლიცირებული storage layer-ი უზრუნველყოფს, რომ ერთი კვანძის დაცემისას <1 წამში ფერიოვერი მოხდება — BMS-ის ოპერატორი ამას ვერც შეამჩნევს. per-region instance უზრუნველყოფს, რომ ევროპელი tenant-ის მონაცემი არასდროს გავა ევროპიდან (GDPR, რეგულაცია).

რა მოხდება ამის გარეშე: single-region Postgres-ი = downtime failover-ზე 5-10 წუთი, მონაცემთა რეზიდენციის დარღვევა GDPR-ის ჯარიმა €20M-მდე ან წლიური ბრუნვის 4%.
TimescaleDB ტელემეტრიის სპეციალიზებული ბაზა — სენსორების მონაცემები (ტემპერატურა, წნევა, ვოლტაჟი) დროის მიხედვით ორგანიზებული ჩვეულებრივი Postgres-ი ვერ გაუმკლავდება წამში 10,000+ ჩანაწერის ჩაწერას ერთ ცხრილში — ხანმოკლე დროში ცხრილი გადიდდება ასეულ გიგაბაიტებამდე და SELECT ... WHERE time > now() - '1h' მოიტანს წუთებს. TimescaleDB ავტომატურად აქცევს ცხრილს hypertable-ად — ყოველი 7 დღის მონაცემი ცალკე chunk-ია, ძველი chunks არ ტვირთავენ ქეშს. Continuous aggregates — 1-საათიანი საშუალოები ავტომატურად ქეშდება, dashboard-ი <100ms-ში იტვირთება მილიონ წერტილზეც. Compression — ძველი მონაცემი 20-ჯერ იკუმშება (20GB → 1GB).

რა მოხდება ამის გარეშე: Postgres-ში 6 თვის მონაცემი შეაგროვებს 500GB-ს, dashboard 30 წამში იტვირთება, ოპერატორი წყვეტს სისტემის გამოყენებას.
Redis (ElastiCache) სესიის ქეში — active org-context, როლები, permissions, 30 წამიანი TTL JWT-ში არ ვდებთ org-context-ს (ADR-017), რადგან: თუ მომხმარებლისთვის როლი შეიცვალა ან membership გაუუქმდა, JWT-ში ჩადებული stale როლი 1 საათი იმოქმედებდა — უსაფრთხოების დარღვევა. Redis-ში inkidental ყოველ request-ზე ვითხოვთ უფრო ახლო truth-ს: 30 წამზე ძველი cache → Aurora-დან refresh. როცა admin ცვლის role-ს → EventBridge ივენტით Redis session immediately მიინდება, შემდეგივე request-ი ახალ როლით მუშაობს.

რა მოხდება ამის გარეშე: revoked ოპერატორი 1 საათამდე კვლავ იღებს dashboard-ს. Compliance breach SOC 2 კონტროლის CC6.3-ში.
S3 (Object Lock) "ცივი" მონაცემების საცავი — ძველი ტელემეტრია Parquet-ში, audit artifacts, 3D assets, PDF ანგარიშები ცხელი მონაცემი (24 საათი) Aurora/Timescale-შია, ცივი — S3-ში. Intelligent-Tiering ავტომატურად გადაანაცვლებს იმ ფაილს, რომელიც 30 დღე არ გამოყენებულა, infrequent-access tier-ზე — ხარჯი 40%-ით მცირდება. Object Lock მოდე = ფაილი მექანიკურადაც ვერ წაიშლება მითითებულ ვადამდე. ეს regulated tenant-ებისთვის სავალდებულოა (ფინანსური მონაცემი 7 წელიწადი უცვლელი უნდა იყოს).

რა მოხდება ამის გარეშე: 100TB ტელემეტრიის შენახვა Aurora-ში = თვეში $20,000; S3 Glacier-ში იგივე = $100. Compliance audit ვერ ჩატარდება, თუ არ შეგვიძლია immutable evidence-ის დადასტურება.
KMS (envelope encryption) გასაღების მენეჯმენტი — ყოველი tenant-ის მონაცემი ცალკე გასაღებით იშიფრება "At-rest encryption"-ი ადვილია: AWS ავტომატურად იცავს diskebs-ი. მაგრამ ერთი default გასაღები ყველა tenant-ისთვის ნიშნავს: insider threat ან compromised IAM role → ყველა tenant-ის monarch-წაკითხვა. Per-tenant CMK = თუ იპოვება tenant A-ს გასაღები, tenant B უსაფრთხოდ რჩება. BYOK (bring-your-own-key) Phase 2-ისთვისაა — regulated tenant-ი თავისი გასაღებით გეტყვის "მხოლოდ ეს KMS გასაღები შეგიძლია გამოიყენო ჩემი მონაცემისთვის".

რა მოხდება ამის გარეშე: ERP tenant-ის ფინანსური მონაცემი გაჟონოს → ჩვენი სხვა 500 tenant-იც compromised ითვლება. Enterprise არ ყიდულობს.
§24.2 — ქსელი და API (Network & API layer)
კომპონენტიროლირატომ საჭიროა (დეტალურად)
CloudFront + WAF + ACM ყველა request-ის პირველი წერტილი — CDN, DDoS დაცვა, TLS terminator CloudFront-ი არის ხიდი მომხმარებელსა და origin API-ს შორის. 200+ edge location-ზე სტატიკური assets-ი დაქეშილია — გვერდი იტვირთება <500ms ნებისმიერი კონტინენტიდან. WAF (Web Application Firewall) იცავს OWASP Top 10 შეტევებისგან: SQL injection, XSS, credential stuffing. თითოეული თავდასხმა მანამდე იჭრება, სანამ API-მდე მიაღწევს. ACM (AWS Certificate Manager) ავტომატურად აახლებს TLS სერტიფიკატებს — ხელით არასდროს ვაკეთებთ certbot renew.

რა მოხდება ამის გარეშე: L7 DDoS შეტევა 10Gbps-ზე = API პაცინული. Expired TLS = ბრაუზერი აჩვენებს "Not secure" და tenant-ი კარგავს ნდობას.
NestJS HTTP API (Fastify) ცენტრალური ბიზნეს ლოგიკა — CRUD, commands, auth, billing, integration endpoints ერთი API = ერთი ადგილი testing-ისთვის, monitoring-ისთვის, version-ინგისთვის. Python Flask legacy-ს ვდეპრეცირებთ (ADR-004), რადგან: 1) TypeScript-ი backend-ში იგივე typing-ს იძლევა frontend-ისთვის — contract-ი ვერანაირად ვერ დადრიფთდება. 2) Fastify 2-ჯერ სწრაფია Express-ზე request/sec-ში. 3) NestJS-ის DI container ნიშნავს: unit test-ი dependency mock-ის გარეშე მიუღებელია — ავტომატურად ენფორსდება არქიტექტურული ჰიგიენა.

რა მოხდება ამის გარეშე: 3 სხვადასხვა API (Flask, NestJS, Lambda) → 3-ჯერ მეტი code duplication, auth bug-ი ერთ-ერთში უჩუმრად იცოცხლებს თვეები.
Socket.io Gateway Real-time კავშირი — dashboard-ის ცოცხალი განახლება, live device status, notifications HTTP polling (ყოველ 5 წამში GET /alerts) — 90% request-ები უფუნქციოა (ახალი მონაცემი არ არის). 1000 tenant × 10 ოპერატორი × 5-წამიანი polling = 2,000 req/sec მუდმივი ტვირთი სიცარიელეზე. WebSocket-ი პერსისტენტული კავშირია — მხოლოდ მაშინ იხარჯება, როცა ახალი მონაცემია. 20× ბანდვიტი, 50× ნაკლები API ტვირთი. Socket.io-ს fallback ladder — თუ WS დაბლოკილია (corporate firewall), ავტომატურად გადადის long-polling-ზე. კლიენტის 100% თავსებადობა.

რა მოხდება ამის გარეშე: ოპერატორი ხედავს alarm-ს 10 წამის დაგვიანებით — critical BMS-ში ეს შეიძლება ტექნიკური ავარიის განსხვავება იყოს.
§24.3 — Device edge (მოწყობილობის ფენა)
კომპონენტიროლირატომ საჭიროა (დეტალურად)
IoT Core (MQTT) მოწყობილობების ცენტრალური "საფოსტო სადგური" — ყველა device აქ აწვება და აქედან იღებს ბრძანებებს Pub/Sub მოდელი ნიშნავს: როცა ქვაბი #3-დან ახალი ტემპერატურის მონაცემი მოდის, ის მიდის ერთდროულად dashboard-ზე, analytics-ზე, alarm rule-ზე — ყოველ subscriber-ზე. HTTP-ში ამდენი fan-out 3× API call-ს საჭიროებდა. mTLS (mutual TLS) — ორივე მხარე ამოწმებს ერთმანეთს სერტიფიკატით. ქვაბი #3-ს აქვს unique cert, რომელიც ACM PCA-ს გამოუშვია. თუ ქურდმა მოიპარა device, სერტიფიკატი ჩვენ შეგვიძლია revoke-ვქნათ — 10 წამში device დაბლოკილია. Retained topics — ახალი subscriber (მაგ. re-connected mobile app) მაშინვე იღებს last-known-state-ს, არ უწევს წუთის ლოდინი შემდეგი მონაცემისთვის.

რა მოხდება ამის გარეშე: HTTP long-polling 10,000 device-ზე = 10,000 persistent connection API-თან, $$$$ cost, ბაზრიდან გასვლის საფრთხე.
Greengrass v2 (edge runtime) ადგილობრივი "გონება" — მოწყობილობის მხარეზე ხდება გადაწყვეტილების მიღება, როცა cloud მიუწვდომელია BMS-ში არის life-safety scenario-ები — მაგ. წყლის ტემპერატურა აიწია 95°C-ზე, boiler unit-ი უნდა გამოირთოს 3 წამში, თორემ აფეთქდება. Cloud-ზე დაყრდნობა ასეთ მომენტში არ შეიძლება: internet connectivity შეიძლება წყდებოდეს, AWS outage მოხდეს, კავშირი 10 წამი დაიკარგოს. Greengrass v2 თვითონ ქვაბის გვერდით ქარვია ჩადგმული — rules-ი აქვე სრულდება, cloud-ის გარეშე. Cloud-ი ხდება observer (ტელემეტრია ვკრიბავ post-factum) და coordinator (კონფიგურაციას ვყიდი device-ებზე), მაგრამ არასდროს critical path-ზე.

რა მოხდება ამის გარეშე: AWS us-east-1 outage → 10,000 boiler-ი უკონტროლოდ → ქონებრივი ზიანი, შესაძლო სასიცოცხლო. UL 60730-ის ცენტრალური მოთხოვნა.
Fleet Provisioning (ACM PCA) ქარხნული "claim" სისტემა — ახალი device ეწოდება, ავტომატურად იღებს per-device სერტიფიკატს Zero-touch enrollment: ქარხანა device-ს ასხამს ერთ "claim" სერტიფიკატს (ყველა მათ შორის), device ჩართვისას ამ claim-ით მიდის IoT Core-ზე, იცვლის თავის უნიკალურ per-device სერტიფიკატზე. ფილდ ოპერატორს არ უწევს device-ების სერიული რიგრიგობით კონფიგურაცია — მაშინს-მოვაგროვდრ-ვდინ. სერტიფიკატი 24 თვეში ვადა ამოუდის, ავტომატურად rotate-ს IoT Core აკეთებს. v1-ში 24 თვე (conservative), Phase 2-ში → 12 თვე (6 თვის stage-proof შემდეგ).

რა მოხდება ამის გარეშე: ხელით provisioning device-ზე = 5 წუთი × 10,000 device = 833 კაცი-საათი. Cost = $$$$, tenant-ი ვერ scale-ავს.

§25.4 — Device onboarding — certification check code
// NestJS onboarding handler — compliance check-ი მოთხოვნილია რეგისტრაციისას
@Post('/v1/orgs/:orgId/devices/register')
async register(@Body() dto: DeviceRegistrationDto) {

  // 1. სავალდებულო ველების ვალიდაცია (zod schema)
  const v = deviceRegistrationSchema.parse(dto);

  // 2. Firmware signature verification (Ed25519 public key per-manufacturer)
  if (!verifyFirmwareSignature(v.firmware_version, v.firmware_signature, v.manufacturer)) {
    throw new UnprocessableEntityException({
      code: 'firmware.signature.invalid',
      detail: 'Firmware signature does not match manufacturer public key'
    });
  }

  // 3. UL/IEC compliance check — tier-based
  const required = ulRequirementsForTier(org.tier);   // enterprise → UL 2900-2-2 sold
  const missing = required.filter(c => !v.certification_marks.includes(c));
  if (missing.length > 0) {
    throw new UnprocessableEntityException({
      code: 'device.certification.missing',
      detail: `Missing required certifications: ${missing.join(', ')}`,
      hint: `Tier '${org.tier}' requires: ${required.join(', ')}`
    });
  }

  // 4. CVE scan against firmware version (nvd.nist.gov API)
  const cves = await cveScanner.scan(v.manufacturer, v.model, v.firmware_version);
  const critical = cves.filter(c => c.severity === 'CRITICAL');
  if (critical.length > 0) {
    throw new UnprocessableEntityException({
      code: 'device.firmware.cve',
      detail: `Firmware has ${critical.length} CRITICAL CVEs`,
      hint: 'Update firmware before registration',
      cves: critical.map(c => c.id)
    });
  }

  // 5. Secure boot attestation (TPM/ATECC signed measurement)
  if (v.boot_mode === 'secure_boot' && !await verifySecureBootAttestation(v)) {
    throw new UnprocessableEntityException({ code: 'device.boot.attestation.failed' });
  }

  // 6. Persist + emit event
  const device = await devices.create({ ...v, tenant_id: org.id, status: 'active' });
  eventBus.emit('device.registered', { device_id: device.id, tenant_id: org.id });
  auditLog.append('device.registered', { actor: req.user.id, entity: device.id, payload: v });

  return { data: device };
}

§28.3 — Scheduling worker (computing next_fire_at in UTC)
// NestJS scheduler worker — runs every minute, picks up next batch
import { CronExpressionParser } from 'cron-parser';
import { zonedTimeToUtc, utcToZonedTime } from 'date-fns-tz';

@Injectable()
export class AutomationScheduler {

  @Cron('* * * * *')   // every minute, look at due rules
  async tick() {
    const now = new Date();
    const due = await this.db.automationRules.findMany({
      where: { status: 'active', next_fire_at_utc: { lte: now } },
      limit: 500
    });

    for (const rule of due) {
      try {
        // 1. dispatch to worker queue (respects quiet_hours, see below)
        await this.queue.send('automation.execute', {
          rule_id: rule.id,
          tenant_id: rule.tenant_id,
          fired_at_utc: now.toISOString()
        });

        // 2. compute NEXT fire — IANA-aware, handles DST
        const nextUtc = computeNextFire(rule.schedule_cron, rule.schedule_tz, now);
        await this.db.automationRules.update(rule.id, {
          last_fired_at: now,
          next_fire_at_utc: nextUtc
        });
      } catch (err) {
        this.otel.recordError(err, { rule_id: rule.id });
        await this.db.automationRules.update(rule.id, { status: 'failed' });
      }
    }
  }
}

function computeNextFire(cron: string, tz: string, from: Date): Date {
  // cron-parser აცნობიერებს IANA tz-ს → IANA ბაზა თვითონ აგვარებს DST-ს
  const iter = CronExpressionParser.parse(cron, {
    currentDate: from,
    tz: tz                          // e.g. 'Asia/Tbilisi'
  });
  return iter.next().toDate();      // returns UTC Date
}

// Quiet hours check — executed inside worker before delivery
function isInQuietHours(userTz: string, quietRange: [number, number] | null): boolean {
  if (!quietRange) return false;
  const localNow = utcToZonedTime(new Date(), userTz);
  const localHour = localNow.getHours();
  const [start, end] = quietRange;
  return start < end
    ? localHour >= start && localHour < end
    : localHour >= start || localHour < end;    // wrap midnight (22-8)
}
